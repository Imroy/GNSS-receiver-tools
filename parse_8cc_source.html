151 int main(int argc, char* argv[]) {
152  std::string filename = "/dev/ttyUSB0";
153  SkyTraqBin::MessageType mt;
154  int rate;
155  bool change_mt = false, change_rate = false;
156 
157  if (argc > 1) {
158  int opt;
159  while ((opt = getopt(argc, argv, "ntbr:")) != -1) {
160  switch (opt) {
161  case 'n':
162  mt = SkyTraqBin::MessageType::None;
163  change_mt = true;
164  break;
165  case 't':
166  mt = SkyTraqBin::MessageType::NMEA0183;
167  change_mt = true;
168  break;
169  case 'b':
170  mt = SkyTraqBin::MessageType::Binary;
171  change_mt = true;
172  break;
173  case 'r':
174  rate = atoi(optarg);
175  change_rate = true;
176  break;
177  default:
178  break;
179  }
180  }
181 
182  if (optind > argc)
183  filename = argv[optind];
184  }
185 
186  std::FILE *file = fopen(filename.c_str(), "a+");
187  if (file == NULL) {
188  std::cerr << "input is not open." << std::endl;
189  exit(1);
190  }
191 
192  auto l = std::make_shared<AppListener>();
193  SkyTraq::Interface iface(file, l);
194 
195  if (change_mt) {
196  std::cout << "Switching to " << mt << " message type..." << std::endl;
197  iface.send(std::make_shared<SkyTraqBin::Config_msg_type>(mt, SkyTraqBin::UpdateType::SRAM),
198  [](bool ack, SkyTraqBin::Output_message* msg) {
199  std::cout << (ack ? "A" : "Not a") << "cknowledged message type switch." << std::endl;
200  });
201  }
202  if (change_rate) {
203  std::cout << "Changing output rate to " << rate << " Hz" << std::endl;
204  iface.send(std::make_shared<SkyTraqBin::Config_sys_pos_rate>(rate, SkyTraqBin::UpdateType::SRAM),
205  [](bool ack, SkyTraqBin::Output_message* msg) {
206  std::cout << (ack ? "A" : "Not a") << "cknowledged output rate change." << std::endl;
207  });
208  }
209 
210  while (1) {
211  try {
212  iface.read();
213  } catch (std::invalid_argument &e) {
214  std::cerr << e.what() << std::endl;
215  } catch (std::exception &e) {
216  std::cerr << e.what() << std::endl;
217  }
218  }
219 
220  return 0;
221 }
