31  unsigned char generate_checksum(std::string tid, std::string type, std::string data) {
32  unsigned char cs;
33  cs = tid[0];
34  cs ^= tid[1];
35 
36  cs ^= type[0];
37  cs ^= type[1];
38  cs ^= type[2];
39 
40  for (auto i : data)
41  cs ^= i;
42 
43  return cs;
44  }

47  std::vector<std::string> split_fields(std::string data) {
48  std::vector<std::string> fields;
49  size_t field_start = data.find_first_of(',') + 1;
50  size_t next_comma;
51 
52  while (field_start != std::string::npos) {
53  next_comma = data.find_first_of(',', field_start);
54  std::string field = data.substr(field_start, next_comma - field_start);
55  fields.push_back(field);
56 
57  if (next_comma == std::string::npos)
58  break;
59  field_start = next_comma + 1;
60  }
61 
62  return fields;
63  }

72  Sentence::ptr parse_sentence(std::string line) {
73  if (line[0] != '$')
74  throw InvalidSentence();
75 
76  size_t first_comma = line.find_first_of(',');
77  std::string tid = line.substr(1, first_comma - 4);
78  std::string type = line.substr(first_comma - 3, 3);
79  std::string data = line.substr(first_comma, line.length() - first_comma - 3);
80 
81  size_t asterisk = line.find_last_of('*');
82  if (asterisk == line.npos)
83  throw InvalidSentence();
84 
85  unsigned char checksum = std::stoi(line.substr(asterisk + 1, line.length() - asterisk - 1), nullptr, 16);
86  unsigned char computed_cs = generate_checksum(tid, type, data);
87  if (computed_cs != checksum)
88  throw ChecksumMismatch(computed_cs, checksum);
89 
90  auto fields = split_fields(data);
91 
92  if ((type == "GGA") &&
93  ((tid == "GP") || (tid == "GN")))
94  return std::make_shared<GGA>(tid, type, fields, checksum);
95 
96  if ((type == "GLL") &&
97  ((tid == "GP") || (tid == "GN")))
98  return std::make_shared<GLL>(tid, type, fields, checksum);
99 
100  if ((type == "GSA") &&
101  ((tid == "GP") || (tid == "GN") || (tid == "BD") || (tid == "GL")))
102  return std::make_shared<GSA>(tid, type, fields, checksum);
103 
104  if ((type == "GSV") &&
105  ((tid == "GP") || (tid == "BD") || (tid == "GL")))
106  return std::make_shared<GSV>(tid, type, fields, checksum);
107 
108  if ((type == "RMC") &&
109  ((tid == "GP") || (tid == "GN")))
110  return std::make_shared<RMC>(tid, type, fields, checksum);
111 
112  if ((type == "VTG") &&
113  ((tid == "GP") || (tid == "GN")))
114  return std::make_shared<VTG>(tid, type, fields, checksum);
115 
116  if ((type == "ZDA") &&
117  ((tid == "GP") || (tid == "GN")))
118  return std::make_shared<ZDA>(tid, type, fields, checksum);
119 
120  if ((type == "STI") &&
121  (tid == "P"))
122  switch (std::stoi(fields[0])) {
123  case 1:
124  return std::make_shared<STI_PPS>(tid, type, fields, checksum);
125 
126  case 4:
127  switch (std::stoi(fields[1])) {
128  case 1:
129  return std::make_shared<STI_sensors>(tid, type, fields, checksum);
130 
131  }
132  }
133 
134  throw UnknownSentenceType(tid, type);
135  }

177  ReceiverMode read_receivermode(std::string field) {
178  ReceiverMode mode = ReceiverMode::unknown;
179 
180  if (field.length() > 0) {
181  if (field == "A")
182  mode = ReceiverMode::Autonomous;
183  else if (field == "D")
184  mode = ReceiverMode::Differential;
185  else if (field == "E")
186  mode = ReceiverMode::Estimated;
187  else if (field == "S")
188  mode = ReceiverMode::Simulated;
189  }
190 
191  return mode;
