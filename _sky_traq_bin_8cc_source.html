32  uint8_t checksum(unsigned char* buffer, Payload_length len) {
33  uint8_t cs = 0;
34  while (len) {
35  cs ^= *buffer;
36  buffer++;
37  len--;
38  }
39 
40  return cs;
41  }

122  Output_message::ptr parse_message(unsigned char* buffer, std::size_t len) {
123  Payload_length payload_len = extract_be<uint16_t>(buffer, 2);
124  std::size_t end = StartSeq_len + PayloadLength_len + payload_len + Checksum_len + EndSeq_len;
125  if (len < end)
126  throw InsufficientData();
127 
128  if ((buffer[0] != 0xa0)
129  || (buffer[1] != 0xa1)
130  || (buffer[end - 2] != 0x0d)
131  || (buffer[end - 1] != 0x0a))
132  throw InvalidMessage();
133 
134  unsigned char *payload = buffer + StartSeq_len + PayloadLength_len;
135  uint16_t id = payload[0];
136  if ((id >= 0x62) && (id <= 0x65)) // construct a composite id if the message has a sub-ID
137  id = (payload[0] << 8) | payload[1];
138 
139  if (output_message_factories.count(id) == 0)
140  throw UnknownMessageID(id);
141 
142  uint8_t cs = buffer[end - 3];
143  {
144  uint8_t ccs = checksum(payload, payload_len);
145  if (cs != ccs)
146  throw ChecksumMismatch(ccs, cs);
147  }
148 
149  return output_message_factories[id](payload, payload_len);
150  }
