31  uint8_t checksum(unsigned char* buffer, Payload_length len) {
32  uint8_t cs = 0;
33  while (len) {
34  cs ^= *buffer;
35  buffer++;
36  len--;
37  }
38 
39  return cs;

120  Output_message::ptr parse_message(unsigned char* buffer, std::size_t len) {
121  Payload_length payload_len = extract_be<uint16_t>(buffer, 2);
122  std::size_t end = 2 + 2 + payload_len + 1 + 2;
123  if (len < end)
124  throw InsufficientData();
125 
126  if ((buffer[0] != 0xa0)
127  || (buffer[1] != 0xa1)
128  || (buffer[end - 2] != 0x0d)
129  || (buffer[end - 1] != 0x0a))
130  throw InvalidMessage();
131 
132  unsigned char *payload = buffer + 2 + 2;
133  uint16_t id = payload[0];
134  if ((id >= 0x62) && (id <= 0x65)) // construct a composite id if the message has a sub-ID
135  id = (payload[0] << 8) | payload[1];
136 
137  if (output_message_factories.count(id) == 0)
138  throw UnknownMessageID(id);
139 
140  uint8_t cs = buffer[end - 3];
141  {
142  uint8_t ccs = checksum(payload, payload_len);
143  if (cs != ccs)
144  throw ChecksumMismatch(ccs, cs);
145  }
146 
147  return output_message_factories[id](payload, payload_len);
148  }
