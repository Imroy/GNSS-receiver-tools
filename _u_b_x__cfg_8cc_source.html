36  Prt::Prt(unsigned char* payload, Length len) :
37  Input_Output_message(Class_ID::CFG, 0x00)
38  {
39  _portid = (PortID)extract_le<uint8_t>(payload, 0);
40  if (len > 1) {
41  uint16_t txReady = extract_le<uint16_t>(payload, 2);
42  _txr_en = txReady & 0x0001;
43  if (_txr_en) {
44  _txr_pol = txReady & 0x0002;
45  _txr_pin = (txReady >> 2) & 0x1f;
46  _txr_thres = txReady >> 7;
47  }
48 
49  uint32_t mode = extract_le<uint32_t>(payload, 4);
50  _mode_charlen = (Mode_Charlen)((mode >> 6) & 0x03);
51  _mode_parity = (Mode_Parity)((mode >> 9) & 0x07);
52  _mode_stopbits = (Mode_Stopbits)((mode >> 12) & 0x03);

54  _baudrate = extract_le<uint32_t>(payload, 8);
55 
56  uint16_t inproto = extract_le<uint16_t>(payload, 12);
57  _inproto_ubx = inproto & 0x0001;

59  _inproto_rtcm = (inproto >> 2) & 0x0001;

70  void Prt::body_to_buf(unsigned char* buffer) const {
71  append_le(buffer, (uint8_t)_portid);
72  append_le<uint8_t>(buffer, 0); // reserved
73 
74  uint16_t txReady = 0;
75  if (_txr_en) {
76  txReady = 1;
77  txReady |= (uint16_t)_txr_pol << 1;
78  txReady |= (uint16_t)(_txr_pin & 0x1f) << 2;
79  txReady |= (_txr_thres & 0x1ff) << 7;
80  }
81  append_le(buffer, txReady);
82 
83  uint32_t mode = 0;
84  mode |= ((uint32_t)_mode_charlen & 0x03) << 6;
85  mode |= ((uint32_t)_mode_parity & 0x07) << 9;
86  mode |= ((uint32_t)_mode_stopbits & 0x03) << 12;
87  append_le(buffer, mode);
88 
89  append_le(buffer, _baudrate);
90  append_le<uint16_t>(buffer, _inproto_ubx | ((uint16_t)_inproto_nmea << 1) | ((uint16_t)_inproto_rtcm << 2));
91  append_le<uint16_t>(buffer, _outproto_ubx | ((uint16_t)_outproto_nmea << 1));
92  append_le<uint16_t>(buffer, (uint16_t)_flags_exttxtimeout << 1);
93  append_le<uint8_t>(buffer, 0); // reserved[2]
94  append_le<uint8_t>(buffer, 0);
95  }
