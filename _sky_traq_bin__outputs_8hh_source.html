623

628  GETTER(uint8_t, PRN, _prn);

630  GETTER(uint8_t*, bytes, _bytes);
631  inline const uint8_t byte(int i) const {
632  if ((i < 0) || (i >= 30))
633  throw std::out_of_range("Can only access byte 0~29");
634  return _bytes[i];

638 
642  template<typename T>
643  const T data(uint8_t start, uint8_t len = sizeof(T) * 8) const {
644  if (start >= 240)
645  throw std::out_of_range("Can only start at bits 0~239");
646  if (start + len >= 240)
647  throw std::out_of_range("Can only end at bits 0~239");
648  if (len > sizeof(T) * 8)
649  throw std::out_of_range("Can only read " + std::to_string(sizeof(T) * 8) + " bits into " + typeid(T).name());
650 
651  // TODO: Make this more efficient by working on whole bytes instead of bits
652  T ret = 0;
653  for (uint8_t from_bit = start, to_bit = 0; to_bit < len; from_bit++, to_bit++) {
654  uint8_t from_byte = from_bit / 8, from_byte_bit = from_bit % 8;
655  int8_t shift = from_byte_bit - to_bit; // right shift; left shift if negative
656 
657  uint8_t mask = 1 << from_byte_bit;
658  if (shift < 0)
