587 
591  template<typename T>
592  const T data(uint8_t start, uint8_t len = sizeof(T) * 8) const {
593  if (start >= 240)
594  throw std::out_of_range("Can only start at bits 0~239");
595  if (start + len >= 240)
596  throw std::out_of_range("Can only end at bits 0~239");
597  if (len > sizeof(T) * 8)
598  throw std::out_of_range("Can only read " + std::to_string(sizeof(T) * 8) + " bits into " + typeid(T).name());
599 
600  // TODO: Make this more efficient by working on whole bytes instead of bits
601  T ret = 0;
602  for (uint8_t from_bit = start, to_bit = 0; to_bit < len; from_bit++, to_bit++) {
603  uint8_t from_byte = from_bit / 8, from_byte_bit = from_bit % 8;
604  int8_t shift = from_byte_bit - to_bit; // right shift; left shift if negative
605 
606  uint8_t mask = 1 << from_byte_bit;
607  if (shift < 0)
608  ret |= (_bytes[from_byte] & mask) << -shift;
609  else
610  ret |= (_bytes[from_byte] & mask) >> shift;
611  }
612 
613 
614  if (std::is_signed<T>::value && (len < sizeof(T) * 8)) {
615  uint8_t i = len - 1;
616  uint8_t mask = 1 << i;
617  if (ret & mask)
618  for (i = len; i <= sizeof(T) * 8; i++)
619  ret |= 1 << i;
620  }
621 
622  return ret;
623  }
