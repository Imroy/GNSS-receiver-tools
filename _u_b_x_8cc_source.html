41  checksum_t checksum(unsigned char* buffer, Length len) {

86  Output_message::ptr parse_message(unsigned char* buffer, std::size_t len) {
87  Length payload_len = extract_le<uint16_t>(buffer, SyncChar_len + ClassID_len);
88  std::size_t end = SyncChar_len + ClassID_len + Length_len + payload_len + Checksum_len;
89  if (len < end)
90  throw GNSS::InsufficientData();
91 
92  if ((buffer[0] != SyncChar[0])
93  || (buffer[1] != SyncChar[1]))
94  throw GNSS::InvalidMessage();
95 
96  checksum_t cs = { buffer[end - 1], buffer[end] };
97  {
98  checksum_t ccs = checksum(buffer + SyncChar_len, ClassID_len + payload_len);
99  if ((cs.A != ccs.A) && (cs.B != ccs.B))
100  throw ChecksumMismatch(ccs, cs);
101  }
102 
103  uint8_t cls = buffer[SyncChar_len], id = buffer[SyncChar_len + 1];
104  uint16_t clsid = ((uint16_t)cls << 8) | id;
105  if (output_message_factories.count(clsid) == 0)
106  throw UnknownMessageID(cls, id);
107 
108  unsigned char *payload = buffer + SyncChar_len + ClassID_len + Length_len;
109 
110  return output_message_factories[clsid](payload, payload_len);
111  }
