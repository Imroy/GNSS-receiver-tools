82  Output_message::ptr parse_message(unsigned char* buffer, std::size_t len) {
83  Length payload_len = extract_le<uint16_t>(buffer, SyncChar_len + ClassID_len);
84  std::size_t end = SyncChar_len + ClassID_len + Length_len + payload_len + Checksum_len;
85  if (len < end)
86  throw GNSS::InsufficientData();
87 
88  if ((buffer[0] != SyncChar[0])
89  || (buffer[1] != SyncChar[1]))
90  throw GNSS::InvalidMessage();
91 
92  checksum_t cs = { buffer[end - 1], buffer[end] };
93  {
94  checksum_t ccs = checksum(buffer + SyncChar_len, ClassID_len + payload_len);
95  if ((cs.A != ccs.A) && (cs.B != ccs.B))
96  throw ChecksumMismatch(ccs, cs);
97  }
98 
99  uint8_t cls = buffer[SyncChar_len], id = buffer[SyncChar_len + 1];
100  uint16_t clsid = ((uint16_t)cls << 8) | id;
101  if (output_message_factories.count(clsid) == 0)
102  throw UnknownMessageID(cls, id);
103 
104  unsigned char *payload = buffer + SyncChar_len + ClassID_len + Length_len;
105 
106  return output_message_factories[clsid](payload, payload_len);
107  }
