56  template<typename T>
57  const T _bits(const uint8_t* bytes, uint8_t start, uint8_t len = sizeof(T) * 8) const {
58  if (start >= 240)
59  throw std::out_of_range("Can only start at bits 0~239");
60  if (start + len >= 240)
61  throw std::out_of_range("Can only end at bits 0~239");
62  if (len > sizeof(T) * 8)
63  throw std::out_of_range("Can only read " + std::to_string(sizeof(T) * 8) + " bits into " + typeid(T).name());
64 
65  // TODO: Make this more efficient by working on whole bytes instead of bits
66  T ret = 0;
67  for (uint8_t from_bit = start, to_bit = 0; to_bit < len; from_bit++, to_bit++) {
68  uint8_t from_byte = from_bit / 8, from_byte_bit = from_bit % 8;
69  int8_t shift = from_byte_bit - to_bit; // right shift; left shift if negative
70 
71  uint8_t mask = 1 << from_byte_bit;
72  if (shift < 0)
73  ret |= (bytes[from_byte] & mask) << -shift;
74  else
75  ret |= (bytes[from_byte] & mask) >> shift;
76  }
77 
78  if (std::is_signed<T>::value && (len < sizeof(T) * 8)) {
79  uint8_t i = len - 1;
80  uint8_t mask = 1 << i;
81  if (ret & mask)
82  for (i = len; i <= sizeof(T) * 8; i++)
83  ret |= 1 << i;
84  }
85 
86  return ret;
87  }
