48  std::vector<Message::ptr> Parser::parse_messages(void) {
49  std::vector<Message::ptr> messages;
50 
51  std::size_t end; // actually the start of the next message
52  do {
53  end = 0;
54  for (std::size_t i = 0; i < _parse_buflen - 1; i++)
55  if ((_parse_buffer[i] == 0x0d)
56  && (_parse_buffer[i + 1] == 0x0a)) {
57  end = i + 2;
58  break;
59  }
60  if (end == 0)
61  break;
62 
63  std::exception_ptr exp = nullptr;
64  try {
65  if (_parse_buffer[0] == '$')
66  messages.push_back(NMEA0183::parse_sentence(std::string((const char*)_parse_buffer, end - 2)));
67  else if ((_parse_buffer[0] == 0xa0) && (_parse_buffer[1] == 0xa1))
68  messages.push_back(SkyTraqBin::parse_message(_parse_buffer, end));
69 
70  // Catch the harmless exceptions
71  } catch (const NMEA0183::InvalidSentence &e) {
72  } catch (const SkyTraqBin::InvalidMessage &e) {
73  } catch (const SkyTraqBin::InsufficientData &e) {
74 
75  } catch (const std::exception) {
76  exp = std::current_exception();
77  }
78 
79  // Remove this packet from the parse buffer
80  if (_parse_buflen > end)
81  memmove(_parse_buffer, _parse_buffer + end, _parse_buflen - end);
82  _parse_buflen -= end;
83  _parse_buffer = (unsigned char*)realloc(_parse_buffer, _parse_buflen);
84 
85  if (exp != nullptr)
86  std::rethrow_exception(exp);
87  } while (_parse_buflen > 0);
88 
89  return messages;
90  }

111  void Interface::read(void) {
112  unsigned char buffer[16];
113  std::size_t len = fread(buffer, 1, 16, _file);
114 
115  if (len == 0) {
116  std::this_thread::sleep_for(std::chrono::milliseconds(10));
117  return;
118  }
119 
120  _parser.add_bytes(buffer, len);
121 
122  auto messages = _parser.parse_messages();
123 
124  for (auto msg : messages) {
125 
126 #define FIRE_IF(class, method) if (msg->isa<class>()) \
127  _listener->method(this, *(msg->cast_as<class>()));
128 
129  try {
130  msg->cast_as<NMEA0183::Sentence>(); // Will throw std::bad_cast if not possible
131 
132  FIRE_IF(NMEA0183::GGA, GGA)
133  else FIRE_IF(NMEA0183::GLL, GLL)
134  else FIRE_IF(NMEA0183::GSA, GSA)
135  else FIRE_IF(NMEA0183::GSV, GSV)
136  else FIRE_IF(NMEA0183::RMC, RMC)
137  else FIRE_IF(NMEA0183::VTG, VTG)
138  else FIRE_IF(NMEA0183::ZDA, ZDA)
139  else FIRE_IF(NMEA0183::STI, STI);
140 
141  } catch (std::bad_cast) {
142  }
143 
144  try {
145  auto m = msg->cast_as<SkyTraqBin::Output_message>(); // Will throw a std::bad_cast exception if not possible
146 
147  if (msg->isa<SkyTraqBin::Ack>()) {
148  auto ack = msg->cast_as<SkyTraqBin::Ack>();
149  uint16_t id = ack->ack_id();
150  if (ack->has_subid())
151  id = (id << 8) | ack->ack_subid();
152 
153  bool response_type = false;
154  uint16_t rid;
155  try {
156  auto *wr = msg->cast_as<SkyTraqBin::with_response>();
157  rid = wr->response_id();
158  if (wr->has_response_subid())
159  rid = (rid << 8) | wr->response_subid();
160  response_type = true;
161  } catch (std::bad_cast) {
162  }
163 
164  if (_response_handlers.count(id) > 0) {
165  // Call the response handler with a null message
166  _response_handlers[id](true, nullptr);
167 
168  // If there is a response message type, move the handler to its id
169  if (response_type)
170  _response_handlers[rid] = _response_handlers[id];
171 
172  _response_handlers.erase(id);
173  }
174 
175  // If there is no additional "response" message incoming, it's okay to send another message
176  if (!response_type)
177  _send_from_queue();
178 
179  } else if (msg->isa<SkyTraqBin::Nack>()) {
180  // There's no additional "response" message, send another message
181  _send_from_queue();
182 
183  auto nack = msg->cast_as<SkyTraqBin::Nack>();
184  uint16_t id = nack->nack_id();
185  if (nack->has_subid())
186  id = (id << 8) | nack->nack_subid();
187 
188  if (_response_handlers.count(id) > 0) {
189  // Call the response handler with a null message
190  _response_handlers[id](false, nullptr);
191  _response_handlers.erase(id);
192  }
193 
194  } else FIRE_IF(SkyTraqBin::Measurement_time, Measurement_time)
195  else FIRE_IF(SkyTraqBin::Raw_measurements, Raw_measurements)
196  else FIRE_IF(SkyTraqBin::SV_channel_status, SV_channel_status)
197  else FIRE_IF(SkyTraqBin::Subframe_data, Subframe_data)
198  else {
199  // Assume this is a "response" ouput message, send another message from the queue
200  _send_from_queue();
201 
202  uint16_t id = m->message_id();
203  try {
204  auto m_with_subid = m->cast_as<SkyTraqBin::with_subid>();
205  id = (id << 8) | m_with_subid->message_subid();
206  } catch (std::bad_cast) {
207  }
208 
209  if (_response_handlers.count(id) > 0) {
210  // Call the response handler with the message
211  _response_handlers[id](true, m);
212  _response_handlers.erase(id);
213  }
214  }
215 
216  } catch (std::bad_cast) {
217  }
218 
219 #undef FIRE_IF
220  }
221  }

224  SkyTraqBin::Payload_length len = msg->message_length();
