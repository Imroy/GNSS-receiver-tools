49  std::vector<Message::ptr> Parser::parse_messages(void) {
50  std::vector<Message::ptr> messages;
51 
52  std::size_t end; // actually the start of the next message
53  do {
54  end = 0;
55  for (std::size_t i = 0; i < _parse_buflen - 1; i++)
56  if ((_parse_buffer[i] == 0x0d)
57  && (_parse_buffer[i + 1] == 0x0a)) {
58  end = i + 2;
59  break;
60  }
61  if (end == 0)
62  break;
63 
64  std::exception_ptr exp = nullptr;
65  try {
66  if (_parse_buffer[0] == '$')
67  messages.push_back(NMEA0183::parse_sentence(std::string((const char*)_parse_buffer, end - 2)));
68  else if ((_parse_buffer[0] == 0xa0) && (_parse_buffer[1] == 0xa1))
69  messages.push_back(SkyTraqBin::parse_message(_parse_buffer, end));
70  else if ((_parse_buffer[0] == 0xb5) && (_parse_buffer[1] = 0x62))
71  messages.push_back(UBX::parse_message(_parse_buffer, end));
72 
73  // Catch the harmless exceptions
74  } catch (const NMEA0183::InvalidSentence &e) {
75  } catch (const GNSS::InvalidMessage &e) {
76  } catch (const GNSS::InsufficientData &e) {
77 
78  } catch (const std::exception) {
79  exp = std::current_exception();
80  }
81 
82  // Remove this packet from the parse buffer
83  if (_parse_buflen > end)
84  memmove(_parse_buffer, _parse_buffer + end, _parse_buflen - end);
85  _parse_buflen -= end;
86  _parse_buffer = (unsigned char*)realloc(_parse_buffer, _parse_buflen);
87 
88  if (exp != nullptr)
89  std::rethrow_exception(exp);
90  } while (_parse_buflen > 0);
91 
92  return messages;
93  }

118  void Interface::read(void) {
119  unsigned char buffer[16];
120  std::size_t len = fread(buffer, 1, 16, _file);
121 
122  if (len == 0) {
123  if (_is_chrdev) {
124  std::this_thread::sleep_for(std::chrono::milliseconds(10));
125  return;
126  } else {
127  throw EndOfFile();
128  }
129  }
130 
131  _parser.add_bytes(buffer, len);
132 
133  auto messages = _parser.parse_messages();
134 
135  for (auto msg : messages) {
136 
137 #define FIRE_IF(class, method) if (msg->isa<class>()) \
138  _listener->method(this, *(msg->cast_as<class>()));
139 
140  try {
141  msg->cast_as<NMEA0183::Sentence>(); // Will throw std::bad_cast if not possible
142 
143  FIRE_IF(NMEA0183::GGA, GGA)
144  else FIRE_IF(NMEA0183::GLL, GLL)
145  else FIRE_IF(NMEA0183::GSA, GSA)
146  else FIRE_IF(NMEA0183::GSV, GSV)
147  else FIRE_IF(NMEA0183::RMC, RMC)
148  else FIRE_IF(NMEA0183::VTG, VTG)
149  else FIRE_IF(NMEA0183::ZDA, ZDA);
150 
151  } catch (std::bad_cast) {
152  }
153 
154  if (typeid(_listener) == typeid(Listener_SkyTraq)) {
155  Listener_SkyTraq *l_s = dynamic_cast<Listener_SkyTraq*>(_listener.get());
156 #undef FIRE_IF
157 #define FIRE_IF(class, method) if (msg->isa<class>()) \
158  l_s->method(this, *(msg->cast_as<class>()));
159 
160  try {
161  msg->cast_as<NMEA0183::Sentence>(); // Will throw std::bad_cast if not possible
162  FIRE_IF(NMEA0183::STI_PPS, STI_PPS)
163  else FIRE_IF(NMEA0183::STI_sensors, STI_sensors);
164  } catch (std::bad_cast) {
165  }
166 
167  try {
168  auto m = msg->cast_as<SkyTraqBin::Output_message>(); // Will throw a std::bad_cast exception if not possible
169 
170  if (msg->isa<SkyTraqBin::Ack>()) {
171  auto ack = msg->cast_as<SkyTraqBin::Ack>();
172  uint16_t id = ack->ack_id();
173  if (ack->has_subid())
174  id = (id << 8) | ack->ack_subid();
175 
176  bool response_type = false;
177  uint16_t rid;
178  try {
179  auto *wr = msg->cast_as<SkyTraqBin::with_response>();
180  rid = wr->response_id();
181  if (wr->has_response_subid())
182  rid = (rid << 8) | wr->response_subid();
183  response_type = true;
184  } catch (std::bad_cast) {
185  }
186 
187  if (_response_handlers_skytraq.count(id) > 0) {
188  // Call the response handler with a null message
189  _response_handlers_skytraq[id](true, nullptr);
190 
191  // If there is a response message type, move the handler to its id
192  if (response_type)
193  _response_handlers_skytraq[rid] = _response_handlers_skytraq[id];
194 
195  _response_handlers_skytraq.erase(id);
196  }
197 
198  // If there is no additional "response" message incoming, it's okay to send another message
199  if (!response_type)
200  _send_from_queue();
201 
202  } else if (msg->isa<SkyTraqBin::Nack>()) {
203  // There's no additional "response" message, send another message
204  _send_from_queue();
205 
206  auto nack = msg->cast_as<SkyTraqBin::Nack>();
207  uint16_t id = nack->nack_id();
208  if (nack->has_subid())
209  id = (id << 8) | nack->nack_subid();
210 
211  if (_response_handlers_skytraq.count(id) > 0) {
212  // Call the response handler with a null message
213  _response_handlers_skytraq[id](false, nullptr);
214  _response_handlers_skytraq.erase(id);
215  }
216 
217  } else FIRE_IF(SkyTraqBin::Nav_data_msg, Navigation_data)
218  else FIRE_IF(SkyTraqBin::Sensor_data, Sensor_data)
219  else FIRE_IF(SkyTraqBin::Measurement_time, Measurement_time)
220  else FIRE_IF(SkyTraqBin::Raw_measurements, Raw_measurements)
221  else FIRE_IF(SkyTraqBin::SV_channel_status, SV_channel_status)
222  else FIRE_IF(SkyTraqBin::Rcv_state, Rcv_state)
223  else FIRE_IF(SkyTraqBin::GPS_subframe_data, GPS_subframe_data)
224  else {
225  // Assume this is a "response" ouput message, send another message from the queue
226  _send_from_queue();
227 
228  uint16_t id = m->message_id();
229  try {
230  auto m_with_subid = m->cast_as<SkyTraqBin::with_subid>();
231  id = (id << 8) | m_with_subid->message_subid();
232  } catch (std::bad_cast) {
233  }
234 
235  if (_response_handlers_skytraq.count(id) > 0) {
236  // Call the response handler with the message
237  _response_handlers_skytraq[id](true, m);
238  _response_handlers_skytraq.erase(id);
239  }
240  }
241 
242  } catch (std::bad_cast) {
243  }
244  }
245 
246 #undef FIRE_IF
247  }
