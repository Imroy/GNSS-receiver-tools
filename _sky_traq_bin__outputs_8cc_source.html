297  Beidou2_subframe_data::Beidou2_subframe_data(unsigned char* payload, Payload_length payload_len) :
298  Output_message(payload, payload_len),
299  _d(payload[0] - 0xE0),
300  _svid(payload[1]),
301  _subframe_num(payload[2])
302  {
303  // Word1: 3(0) ~ 6(1)
304  _words[0] = static_cast<uint32_t>(payload[3])
305  | (static_cast<uint32_t>(payload[4]) << 8)
306  | (static_cast<uint32_t>(payload[5]) << 16)
307  | ((static_cast<uint32_t>(payload[6]) & 0x03) << 24);
308 
309  // Word2: 6(2) ~ 8(7)
310  _words[1] = (static_cast<uint32_t>(payload[6]) >> 2)
311  | (static_cast<uint32_t>(payload[7]) << 6)
312  | (static_cast<uint32_t>(payload[8]) << 14);
313 
314  for (uint8_t i = 0; i < 8; i += 4) {
315  uint8_t b = 9 + ((i * 22) >> 3);
316 
317  // Word3: 9(0) ~ 11(5)
318  // Word7: 20(0) ~ 22(5)
319  _words[i+2] = static_cast<uint32_t>(payload[b])
320  | (static_cast<uint32_t>(payload[b+1]) << 8)
321  | ((static_cast<uint32_t>(payload[b+2]) & 0x3f) << 16);
322 
323  // Word4: 11(6) ~ 14(3)
324  // Word8: 22(6) ~ 25(3)
325  _words[i+3] = (static_cast<uint32_t>(payload[b+2]) >> 6)
326  | (static_cast<uint32_t>(payload[b+3]) << 2)
327  | (static_cast<uint32_t>(payload[b+4]) << 10)
328  | ((static_cast<uint32_t>(payload[b+5]) & 0x0f) << 18);
329 
330  // Word5: 14(4) ~ 17(1)
331  // Word9: 25(4) ~ 28(1)
332  _words[i+4] = (static_cast<uint32_t>(payload[b+5]) >> 4)
333  | (static_cast<uint32_t>(payload[b+6]) << 4)
334  | (static_cast<uint32_t>(payload[b+7]) << 12)
335  | ((static_cast<uint32_t>(payload[b+8]) & 0x03) << 20);
336 
337  // Word6: 17(2) ~ 19(7)
338  // Word10: 28(2) ~ 30(7)
339  _words[i+5] = (static_cast<uint32_t>(payload[b+8]) >> 2)
340  | (static_cast<uint32_t>(payload[b+9]) << 6)
341  | (static_cast<uint32_t>(payload[b+10]) << 14);
342  }
343  }
